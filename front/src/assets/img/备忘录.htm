js基础 arguments 面试 
var length = 10;

function fn () {
    console.log(this.length);
}

var obj = {
    length: 5,
    method: function(fn) {
        fn();
        /* arguments 是一个数组，第一个参数为函数 fn，第二个参数为 1 */
        arguments[0]();
        /* arguments[0]() 相当于是向 arguments 对象调用方法 */
        /* 因此该方法里面的 this 指向的是 arguments, arguments 对象有两个参数，即 length = 2 */
    }
}

obj.method(fn, 1);
/* 相当于 ==> obj.method(window.fn, 1) */
/* obj.method(window.fn, 1); */
